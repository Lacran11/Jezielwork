$('input[type="radio"]').change(function() {
    var classes = $(this).prop('class').split(' '); // Obtener las clases del radio button
    var colClass = classes.find(c => c.startsWith('c')); // Columna
    var rowClass = classes.find(c => c.startsWith('r')); // Fila

    // Si un radio button es seleccionado
    var selectedValue = $(this).val();

    if (selectedValue) {
        // Desmarcar otros radio buttons en la misma fila
        $('input.' + rowClass).not(this).prop('checked', false);

        // Desmarcar otros radio buttons en la misma columna
        $('input.' + colClass).not(this).prop('checked', false);
    }
});

// Función para validar la selección
$('#validateBtn').click(function() {
    var selectedItems = 0;

    // Contamos cuántos radio buttons están seleccionados
    $('input[type="radio"]:checked').each(function() {
        selectedItems++;
    });

    // Validación: Si hay exactamente 4 elementos seleccionados
    if (selectedItems === 4) {
        $('#validationMessage').text('¡Selección correcta!').css('color', 'green');
    } else {
        $('#validationMessage').text('Por favor, selecciona exactamente 4 elementos.').css('color', 'red');
    }
});
